    <script>
        // PSS-10 Questions
        const questions = [
            { 
                id: 1, 
                text: "No último mês, com que frequência você ficou aborrecido por causa de algo que aconteceu inesperadamente?",
                type: "negative"
            },
            { 
                id: 2, 
                text: "No último mês, com que frequência você sentiu que não conseguia controlar as coisas importantes em sua vida?",
                type: "negative"
            },
            { 
                id: 3, 
                text: "No último mês, com que frequência você se sentiu nervoso e estressado?",
                type: "negative"
            },
            { 
                id: 4, 
                text: "No último mês, com que frequência você sentiu confiança na sua habilidade para resolver seus problemas pessoais?",
                type: "positive"
            },
            { 
                id: 5, 
                text: "No último mês, com que frequência você sentiu que as coisas estavam acontecendo de acordo com a sua vontade?",
                type: "positive"
            },
            { 
                id: 6, 
                text: "No último mês, com que frequência você achou que não conseguiria lidar com todas as coisas que você tinha que fazer?",
                type: "negative"
            },
            { 
                id: 7, 
                text: "No último mês, com que frequência você conseguiu controlar as irritações em sua vida?",
                type: "positive"
            },
            { 
                id: 8, 
                text: "No último mês, com que frequência você sentiu que tudo estava sob controle?",
                type: "positive"
            },
            { 
                id: 9, 
                text: "No último mês, com que frequência você ficou irritado porque as coisas que aconteceram estavam fora do seu controle?",
                type: "negative"
            },
            { 
                id: 10, 
                text: "No último mês, com que frequência você sentiu que as dificuldades se acumularam a ponto de você não conseguir superá-las?",
                type: "negative"
            }
        ];

        const options = [
            { value: 0, text: "Nunca" },
            { value: 1, text: "Quase nunca" },
            { value: 2, text: "Às vezes" },
            { value: 3, text: "Frequentemente" },
            { value: 4, text: "Muito frequentemente" }
        ];

        let currentQuestion = 0;
        let answers = {};

        // Initialize test
        function initTest() {
            renderQuestion();
            updateProgress();
            updateNavigation();
        }

        // Render current question
        function renderQuestion() {
            const testForm = document.getElementById('testForm');
            const question = questions[currentQuestion];
            const isPositiveQuestion = question.type === 'positive';
            
            testForm.innerHTML = `
                <div class="question-card bg-white rounded-2xl p-8 ${answers[question.id] !== undefined ? 'answered' : ''}">
                    <div class="mb-6">
                        <span class="inline-block ${isPositiveQuestion ? 'bg-green-100 text-green-800' : 'bg-red-100 text-red-800'} text-sm font-medium px-3 py-1 rounded-full mb-4">
                            ${isPositiveQuestion ? '😊 Enfrentamento Positivo' : '😰 Situação Estressante'}
                        </span>
                        <h3 class="text-xl font-semibold text-gray-800 leading-relaxed">
                            ${currentQuestion + 1}. ${question.text}
                        </h3>
                        <p class="text-gray-600 mt-3 text-sm">
                            ${isPositiveQuestion ? 
                                '💡 Esta pergunta avalia sua capacidade de lidar com situações difíceis.' : 
                                '⚠️ Esta pergunta avalia situações que podem gerar stress.'
                            }
                        </p>
                    </div>
                    
                    <div class="space-y-3">
                        ${options.map(option => `
                            <label class="radio-option flex items-center p-4 rounded-lg cursor-pointer border border-gray-200 hover:bg-red-50">
                                <input type="radio" name="question_${question.id}" value="${option.value}" 
                                       class="w-4 h-4 text-red-600 border-gray-300 focus:ring-red-500"
                                       ${answers[question.id] === option.value ? 'checked' : ''}
                                       onchange="handleAnswer(${question.id}, ${option.value})">
                                <span class="ml-3 text-gray-700 font-medium">${option.text}</span>
                            </label>
                        `).join('')}
                    </div>
                    
                    ${isPositiveQuestion ? `
                        <div class="mt-4 p-3 bg-green-50 border border-green-200 rounded-lg">
                            <p class="text-green-700 text-sm">
                                ✅ Respostas com maior frequência são positivas para esta questão
                            </p>
                        </div>
                    ` : `
                        <div class="mt-4 p-3 bg-red-50 border border-red-200 rounded-lg">
                            <p class="text-red-700 text-sm">
                                ⚠️ Respostas com maior frequência indicam maior stress para esta questão
                            </p>
                        </div>
                    `}
                </div>
            `;
        }

        // Handle answer selection
        function handleAnswer(questionId, value) {
            answers[questionId] = value;
            updateProgress();
            updateNavigation();
            
            // Add visual feedback
            const questionCard = document.querySelector('.question-card');
            questionCard.classList.add('answered');
        }

        // Update progress bar
        function updateProgress() {
            const progress = (Object.keys(answers).length / questions.length) * 100;
            document.getElementById('progressBar').style.width = `${progress}%`;
            document.getElementById('progressText').textContent = `Pergunta ${currentQuestion + 1} de ${questions.length}`;
        }

        // Update navigation buttons
        function updateNavigation() {
            const prevBtn = document.getElementById('prevBtn');
            const nextBtn = document.getElementById('nextBtn');
            
            prevBtn.disabled = currentQuestion === 0;
            
            if (currentQuestion === questions.length - 1) {
                nextBtn.textContent = Object.keys(answers).length === questions.length ? '📊 Ver Resultados' : 'Finalizar';
                nextBtn.disabled = answers[questions[currentQuestion].id] === undefined;
            } else {
                nextBtn.textContent = 'Próxima →';
                nextBtn.disabled = answers[questions[currentQuestion].id] === undefined;
            }
        }

        // Navigation functions
        function prevQuestion() {
            if (currentQuestion > 0) {
                currentQuestion--;
                renderQuestion();
                updateProgress();
                updateNavigation();
            }
        }

        function nextQuestion() {
            if (currentQuestion < questions.length - 1) {
                currentQuestion++;
                renderQuestion();
                updateProgress();
                updateNavigation();
            } else {
                showResults();
            }
        }

        // Calculate and show results
        function showResults() {
            // Calculate total score with reverse scoring for positive items
            let totalScore = 0;
            
            questions.forEach(question => {
                const answer = answers[question.id];
                if (question.type === 'positive') {
                    // Reverse score positive items (4-answer)
                    totalScore += (4 - answer);
                } else {
                    // Direct score negative items
                    totalScore += answer;
                }
            });
            
            // Calculate separate scores for analysis
            const negativeScore = questions
                .filter(q => q.type === 'negative')
                .reduce((sum, q) => sum + answers[q.id], 0);
            
            const positiveScore = questions
                .filter(q => q.type === 'positive')
                .reduce((sum, q) => sum + (4 - answers[q.id]), 0);
            
            let severity = '';
            let color = '';
            let recommendation = '';
            let description = '';
            let stressLevel = '';

            // PSS-10 interpretation (0-40 scale)
            if (totalScore >= 0 && totalScore <= 13) {
                severity = 'Stress Baixo';
                stressLevel = 'Baixo';
                color = 'green';
                description = 'Níveis normais de stress percebido';
                recommendation = 'Você está lidando bem com as situações do dia a dia. Continue mantendo suas estratégias de enfrentamento.';
            } else if (totalScore >= 14 && totalScore <= 26) {
                severity = 'Stress Moderado';
                stressLevel = 'Moderado';
                color = 'yellow';
                description = 'Níveis moderados de stress percebido';
                recommendation = 'Seu nível de stress está um pouco elevado. Considere técnicas de relaxamento, exercícios físicos e organização da rotina.';
            } else if (totalScore >= 27 && totalScore <= 40) {
                severity = 'Stress Alto';
                stressLevel = 'Alto';
                color = 'red';
                description = 'Níveis elevados de stress percebido';
                recommendation = 'Você está enfrentando níveis altos de stress. É importante buscar estratégias de manejo do stress e considerar ajuda profissional.';
            }

            document.getElementById('testForm').classList.add('hidden');
            document.getElementById('navigationButtons').classList.add('hidden');
            
            const resultsSection = document.getElementById('resultsSection');
            const resultContent = document.getElementById('resultContent');
            
            resultContent.innerHTML = `
                <div class="text-center mb-8">
                    <div class="inline-block p-6 bg-${color}-100 rounded-full mb-4">
                        <span class="text-4xl">
                            ${color === 'green' ? '😌' : color === 'yellow' ? '😐' : '😰'}
                        </span>
                    </div>
                    <h3 class="text-2xl font-bold text-gray-800 mb-2">${severity}</h3>
                    <p class="text-gray-600 mb-2">${description}</p>
                    <div class="text-3xl font-bold text-${color}-600">${totalScore}/40 pontos</div>
                </div>
                
                <div class="bg-${color}-50 p-6 rounded-xl mb-8">
                    <h4 class="text-lg font-semibold text-${color}-800 mb-3">💡 Recomendações</h4>
                    <p class="text-${color}-700 mb-4">${recommendation}</p>
                    
                    ${color !== 'green' ? `
                        <div class="bg-white p-4 rounded-lg">
                            <h5 class="font-semibold text-gray-800 mb-2">🛠️ Estratégias de Manejo do Stress:</h5>
                            <ul class="text-sm text-gray-700 space-y-1">
                                <li>• Técnicas de respiração e relaxamento</li>
                                <li>• Exercícios físicos regulares</li>
                                <li>• Organização do tempo e prioridades</li>
                                <li>• Sono adequado (7-9 horas por noite)</li>
                                <li>• Apoio social e comunicação</li>
                                <li>• Atividades prazerosas e hobbies</li>
                                ${color === 'red' ? '<li>• Considerar acompanhamento psicológico</li>' : ''}
                            </ul>
                        </div>
                    ` : ''}
                </div>
                
                <div class="grid md:grid-cols-2 gap-6 mb-8">
                    <div class="bg-red-50 p-6 rounded-xl">
                        <h4 class="text-lg font-semibold text-red-800 mb-2">😰 Situações Estressantes</h4>
                        <div class="text-3xl font-bold text-red-600">${negativeScore}/24</div>
                        <p class="text-sm text-red-700">pontos de stress negativo</p>
                        <div class="mt-2 bg-red-200 rounded-full h-2">
                            <div class="bg-red-600 h-2 rounded-full" style="width: ${(negativeScore/24)*100}%"></div>
                        </div>
                    </div>
                    
                    <div class="bg-green-50 p-6 rounded-xl">
                        <h4 class="text-lg font-semibold text-green-800 mb-2">😊 Capacidade de Enfrentamento</h4>
                        <div class="text-3xl font-bold text-green-600">${16-positiveScore}/16</div>
                        <p class="text-sm text-green-700">pontos de enfrentamento</p>
                        <div class="mt-2 bg-green-200 rounded-full h-2">
                            <div class="bg-green-600 h-2 rounded-full" style="width: ${((16-positiveScore)/16)*100}%"></div>
                        </div>
                    </div>
                </div>
                
                <div class="bg-gray-50 p-6 rounded-xl mb-6">
                    <h4 class="text-lg font-semibold text-gray-800 mb-4">📊 Interpretação dos Resultados</h4>
                    <div class="grid grid-cols-1 md:grid-cols-3 gap-4 text-sm">
                        <div class="text-center p-4 bg-green-100 rounded-lg">
                            <div class="font-bold text-green-800">0-13</div>
                            <div class="text-green-600">Stress Baixo</div>
                            <div class="text-xs text-green-500 mt-1">Enfrentamento adequado</div>
                        </div>
                        <div class="text-center p-4 bg-yellow-100 rounded-lg">
                            <div class="font-bold text-yellow-800">14-26</div>
                            <div class="text-yellow-600">Stress Moderado</div>
                            <div class="text-xs text-yellow-500 mt-1">Atenção necessária</div>
                        </div>
                        <div class="text-center p-4 bg-red-100 rounded-lg">
                            <div class="font-bold text-red-800">27-40</div>
                            <div class="text-red-600">Stress Alto</div>
                            <div class="text-xs text-red-500 mt-1">Intervenção recomendada</div>
                        </div>
                    </div>
                    <div class="mt-4 p-3 bg-blue-50 rounded-lg">
                        <p class="text-sm text-blue-700">
                            <strong>Seu resultado:</strong> ${totalScore} pontos - ${severity}
                        </p>
                    </div>
                </div>
                
                <div class="bg-gray-50 p-6 rounded-xl">
                    <h4 class="text-lg font-semibold text-gray-800 mb-4">📈 Análise Detalhada</h4>
                    <div class="space-y-2 text-sm text-gray-700">
                        <p><strong>Teste aplicado:</strong> PSS-10 (Perceived Stress Scale - 10 items)</p>
                        <p><strong>Data da avaliação:</strong> ${new Date().toLocaleDateString('pt-BR')}</p>
                        <p><strong>Período avaliado:</strong> Último mês</p>
                        <p><strong>Pontuação total:</strong> ${totalScore}/40 pontos</p>
                        <p><strong>Nível de stress:</strong> ${stressLevel}</p>
                        <p><strong>Stress situacional:</strong> ${negativeScore}/24 pontos</p>
                        <p><strong>Capacidade de enfrentamento:</strong> ${16-positiveScore}/16 pontos</p>
                    </div>
                </div>
                
                ${color === 'red' ? `
                    <div class="mt-6 p-6 bg-red-50 border-2 border-red-200 rounded-xl">
                        <h4 class="text-lg font-bold text-red-800 mb-3">🚨 Ação Recomendada</h4>
                        <p class="text-red-700 mb-3">
                            Seus níveis de stress estão elevados. Considere as seguintes ações:
                        </p>
                        <ul class="list-disc list-inside text-red-700 space-y-1">
                            <li>Procurar um psicólogo para técnicas de manejo do stress</li>
                            <li>Consultar um médico para avaliar impactos físicos</li>
                            <li>Implementar técnicas de relaxamento diariamente</li>
                            <li>Revisar sua rotina e reduzir sobrecarga quando possível</li>
                            <li>Buscar apoio social de familiares e amigos</li>
                        </ul>
                    </div>
                ` : ''}
            `;
            
            resultsSection.classList.remove('hidden');
        }

        // Reset test
        function resetTest() {
            currentQuestion = 0;
            answers = {};
            document.getElementById('testForm').classList.remove('hidden');
            document.getElementById('navigationButtons').classList.remove('hidden');
            document.getElementById('resultsSection').classList.add('hidden');
            initTest();
        }

        // Go to assessment center
        function goToCenter() {
            window.location.href = '/centro-avaliacao';
        }

        // Event listeners
        document.getElementById('prevBtn').addEventListener('click', prevQuestion);
        document.getElementById('nextBtn').addEventListener('click', nextQuestion);

        // Initialize the test
        initTest();
    </script>
