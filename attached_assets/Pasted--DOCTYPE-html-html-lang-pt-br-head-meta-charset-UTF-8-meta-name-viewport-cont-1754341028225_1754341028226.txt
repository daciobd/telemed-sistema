  <script>
    let patientResponse = null;
    let pacienteIdRemarcar = null;

    function atualizarAgenda() {
      const data = document.getElementById('dataSelecionada').value;
      alert(`Agenda atualizada para ${data}. Integre com o banco para carregar pacientes.`);
    }

    function invitePatient(patientId, patientName) {
      patientResponse = null;
      alert(`Convite enviado para ${patientName} para a teleconsulta às ${new Date().toLocaleTimeString()}. Aguardando resposta...`);
      
      setTimeout(() => {
        if (!patientResponse) {
          if (confirm(`O paciente ${patientName} não respondeu ao convite. Deseja remarcar, cancelar ou insistir?`)) {
            if (confirm("Remarcar consulta?")) {
              abrirAgenda(patientId);
            } else {
              cancelarConsulta(patientId);
            }
          } else {
            alert("Insistindo no convite...");
            invitePatient(patientId, patientName);
          }
        }
      }, 5000);
    }

    function startConsulta(patientId, event) {
      event.stopPropagation();
      window.location.href = `consulta.html?id=${patientId}`;
    }

    function cancelarConsulta(patientId, event) {
      event?.stopPropagation();
      if (confirm(`Deseja cancelar a consulta do paciente ID ${patientId}?`)) {
        alert(`Consulta do paciente ID ${patientId} cancelada.`);
      }
    }

    function abrirAgenda(patientId, event) {
      event?.stopPropagation();
      pacienteIdRemarcar = patientId;
      document.getElementById('modalAgenda').style.display = 'block';

      // Simulação de horários disponíveis (substituir por banco de dados)
      const horarios = [
        { hora: '09:00', disponivel: true },
        { hora: '10:30', disponivel: false }, // Ocupado
        { hora: '14:00', disponivel: true },
        { hora: '15:30', disponivel: true }
      ];

      const agenda = document.getElementById('horariosDisponiveis');
      agenda.innerHTML = horarios.map(h => `
        <div class="horario ${!h.disponivel ? 'ocupado' : ''}" onclick="${h.disponivel ? `remarcarConsulta(${patientId}, '${h.hora}')` : ''}">
          ${h.hora} ${!h.disponivel ? '(Ocupado)' : ''}
        </div>
      `).join('');
    }

    function fecharAgenda() {
      document.getElementById('modalAgenda').style.display = 'none';
    }

    function remarcarConsulta(patientId, novoHorario) {
      alert(`Consulta do paciente ID ${patientId} remarcada para ${novoHorario} de ${document.getElementById('dataSelecionada').value || 'data selecionada'}.`);
      fecharAgenda();
      // Em produção, atualize o banco de dados aqui
    }

    window.addEventListener('click', () => {
      if (!patientResponse) patientResponse = true;
    });
  </script>
