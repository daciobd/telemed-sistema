Correção definitiva (evita colapsar de novo)

Aplique estas 3 mudanças rápidas no seu enhanced-consultation.html (ou no JS do enhanced):

1) CSS – limite mínimo e visual do divisor
:root{
  --notes-min: 360px;
  --notes-max: 760px;
  --notes-default: 420px;
}
.notes-panel{ min-width: var(--notes-min); }
.resizer{
  width: 8px; cursor: col-resize; background: rgba(255,255,255,.06);
  border-left: 1px solid rgba(255,255,255,.12);
}
.resizer:hover{ background: rgba(255,255,255,.12); }

2) JS – “clamp”, persistência segura e reset por duplo clique
<script>
(function(){
  const NOTES_KEY = 'telemed_notes_width';
  const MIN = 360, MAX = 760, DEF = 420;

  const notes = document.querySelector('.notes-panel');
  const resizer = document.querySelector('.resizer');

  const clamp = (v,min,max) => Math.max(min, Math.min(max, v));
  const px = n => `${n}px`;

  // largura inicial segura
  let w = parseInt(localStorage.getItem(NOTES_KEY) || DEF, 10);
  w = isNaN(w) ? DEF : clamp(w, MIN, MAX);
  if (notes) notes.style.width = px(w);

  // drag
  let startX = 0, startW = 0, dragging = false;
  function onMove(e){
    if(!dragging) return;
    const dx = e.clientX - startX;
    const nw = clamp(startW + dx, MIN, MAX);
    notes.style.width = px(nw);
  }
  function onUp(){
    if(!dragging) return;
    dragging = false;
    const final = parseInt(getComputedStyle(notes).width,10);
    localStorage.setItem(NOTES_KEY, clamp(final, MIN, MAX));
    window.removeEventListener('mousemove', onMove);
    window.removeEventListener('mouseup', onUp);
  }
  resizer?.addEventListener('mousedown', (e)=>{
    dragging = true;
    startX = e.clientX;
    startW = parseInt(getComputedStyle(notes).width,10);
    window.addEventListener('mousemove', onMove);
    window.addEventListener('mouseup', onUp);
  });

  // duplo clique no divisor = reset de layout
  resizer?.addEventListener('dblclick', ()=>{
    notes.style.width = px(DEF);
    localStorage.setItem(NOTES_KEY, DEF);
  });

  // util de depuração
  window.telemedEnhancedDebug = Object.assign(window.telemedEnhancedDebug||{}, {
    resetLayout(){ localStorage.removeItem(NOTES_KEY); location.reload(); }
  });
})();
</script>

3) (Opcional) Botão de reset visível

Coloque perto do divisor ou no topo das anotações:

<button onclick="telemedEnhancedDebug.resetLayout()" title="Redefinir layout">
  Reset UI
</button>
