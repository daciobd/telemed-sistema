    <script>
        function filterPatients() {
            const id = document.getElementById('filter-id').value.toLowerCase();
            const nome = document.getElementById('filter-nome').value.toLowerCase();
            const sobrenome = document.getElementById('filter-sobrenome').value.toLowerCase();
            const mae = document.getElementById('filter-mae').value.toLowerCase();
            const identificacao = document.getElementById('filter-identificacao').value.toLowerCase();
            const especialidade = document.getElementById('filter-especialidade').value.toLowerCase();

            const rows = document.querySelectorAll('#patients-table tbody tr');
            rows.forEach(row => {
                const cells = row.querySelectorAll('td');
                const rowId = cells[0].textContent.toLowerCase();
                const rowNome = cells[1].textContent.toLowerCase();
                const rowIdent = cells[2].textContent.toLowerCase();
                const rowEspecialidade = row.dataset.especialidade.toLowerCase();

                // Simples filtro (adapte para mais campos se necessário)
                const matchId = rowId.includes(id);
                const matchNome = rowNome.includes(nome) || rowNome.includes(sobrenome) || rowNome.includes(mae);
                const matchIdent = rowIdent.includes(identificacao);
                const matchEsp = especialidade === '' || rowEspecialidade.includes(especialidade);

                row.style.display = (matchId && matchNome && matchIdent && matchEsp) ? '' : 'none';
            });
        }
    </script>
