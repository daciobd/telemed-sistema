  <script>
    let patientResponse = null;

    function invitePatient(patientId, patientName) {
      patientResponse = null; // Reseta a resposta
      alert(`Convite enviado para ${patientName} para a teleconsulta às ${new Date().toLocaleTimeString()}. Aguardando resposta...`);
      
      // Simula atraso na resposta do paciente (5 segundos)
      setTimeout(() => {
        if (!patientResponse) {
          if (confirm(`O paciente ${patientName} não respondeu ao convite. Deseja remarcar, cancelar ou insistir?`)) {
            // O médico decide: true = remarcar, false = cancelar
            if (confirm("Remarcar consulta?")) {
              remarcarConsulta(patientId);
            } else {
              cancelarConsulta(patientId);
            }
          } else {
            alert("Insistindo no convite...");
            invitePatient(patientId, patientName); // Tenta novamente
          }
        }
      }, 5000);
    }

    function startConsulta(patientId, event) {
      event.stopPropagation(); // Evita o clique na linha
      window.location.href = `consulta.html?id=${patientId}`;
    }

    function remarcarConsulta(patientId, event) {
      event?.stopPropagation();
      alert(`Consulta do paciente ID ${patientId} remarcada para outra data.`);
      // Aqui você pode integrar com o banco para atualizar o horário
    }

    function cancelarConsulta(patientId, event) {
      event?.stopPropagation();
      if (confirm(`Deseja cancelar a consulta do paciente ID ${patientId}?`)) {
        alert(`Consulta do paciente ID ${patientId} cancelada.`);
        // Integre com o banco para remover ou marcar como cancelado
      }
    }

    // Simulação de resposta do paciente (para teste, remova em produção)

      if (!patientResponse) patientResponse = true; // Simula aceitação
    });
