<script>
function createBreadcrumb(path) {
    const breadcrumbData = {
        'dashboard': { icon: '🏠', name: 'Dashboard' },
        'dr-ai': { icon: '🤖', name: 'Dr. AI' },
        'triagem': { icon: '🩺', name: 'Triagem Inteligente' },
        'protocolos': { icon: '📋', name: 'Protocolos Clínicos' },
        'prescricoes': { icon: '💊', name: 'Prescrições' },
        'pdf': { icon: '📄', name: 'Gerador de PDF' },
        'avaliacao': { icon: '🧠', name: 'Centro de Avaliação' },
        'gad7': { icon: '😰', name: 'GAD-7 - Ansiedade' },
        'phq9': { icon: '😔', name: 'PHQ-9 - Depressão' }
    };
    
    let breadcrumbHTML = '<div class="breadcrumb-container"><nav class="breadcrumb">';
    
    path.forEach((item, index) => {
        const data = breadcrumbData[item];
        const isLast = index === path.length - 1;
        
        if (isLast) {
            breadcrumbHTML += `
                <div class="breadcrumb-item active">
                    <span class="breadcrumb-icon">${data.icon}</span>
                    ${data.name}
                </div>
            `;
        } else {
            breadcrumbHTML += `
                <div class="breadcrumb-item">
                    <a href="javascript:history.back()">
                        <span class="breadcrumb-icon">${data.icon}</span>
                        ${data.name}
                    </a>
                </div>
                <span class="breadcrumb-separator">›</span>
            `;
        }
    });
    
    breadcrumbHTML += '</nav></div>';
    return breadcrumbHTML;
}

</script>
