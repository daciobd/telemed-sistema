// Função para mostrar mensagens aprimoradas
function showEnhancedMessage(section) {
    const message = ENHANCED_DEMO_MESSAGES[section];
    
    if (!message) {
        // Fallback para seções não mapeadas
        return showGenericMessage(section);
    }
    
    // Mostrar loading primeiro
    showCustomLoading(message.loading);
    
    // Depois de 2 segundos, mostrar conteúdo completo
    setTimeout(() => {
        hideCustomLoading();
        showCustomAlert(message.title, message.content);
    }, 2000);
}

                @keyframes spin { 
                    to { transform: rotate(360deg); } 
                }
                @keyframes loading-bar {
                    to { transform: translateX(0); }

            </style>
        </div>
    `;
    
    document.body.insertAdjacentHTML('beforeend', loadingHTML);
}

function hideCustomLoading() {
    const loading = document.getElementById('custom-loading');
    if (loading) {
        loading.style.opacity = '0';
        loading.style.transition = 'opacity 0.3s ease';
        setTimeout(() => loading.remove(), 300);
    }
}
