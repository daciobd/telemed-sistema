function generatePDF(testName, results) {
    const { jsPDF } = window.jspdf;

    // Configura√ß√µes

    // Fun√ß√£o para adicionar texto com quebra de linha
    function addText(text, x, y, maxWidth) {

        doc.text(lines, x, y);
        return y + (lines.length * lineHeight);
    }
    
    // Cabe√ßalho

    doc.setFont(undefined, 'bold');
    doc.text('TeleMed Pro - Centro de Avalia√ß√£o Psiqui√°trica', margin, yPosition);

    // T√≠tulo do teste

    // Data

    doc.setFont(undefined, 'normal');
    doc.text(`Data: ${new Date().toLocaleDateString('pt-BR')}`, margin, yPosition);
    yPosition += 15;
    
    // Resultados principais
    doc.setFont(undefined, 'bold');
    doc.text('RESULTADOS:', margin, yPosition);
    yPosition += 10;
    
    doc.setFont(undefined, 'normal');
    yPosition = addText(`Pontua√ß√£o Total: ${results.totalScore}/${results.maxScore}`, margin, yPosition, pageWidth - 2*margin);

    // Interpreta√ß√£o

    // Recomenda√ß√µes
    doc.setFont(undefined, 'bold');
    doc.text('RECOMENDA√á√ïES:', margin, yPosition);

    doc.setFont(undefined, 'normal');
    yPosition = addText(results.recommendations, margin, yPosition, pageWidth - 2*margin);

    // Detalhes por categoria (se existir)
