    <script>
        // GAD-7 Questions
        const questions = [

            { id: 2, text: "N√£o conseguir parar ou controlar as preocupa√ß√µes" },
            { id: 3, text: "Preocupar-se muito com diversas coisas" },
            { id: 4, text: "Dificuldade para relaxar" },
            { id: 5, text: "Ficar t√£o agitado que se torna dif√≠cil permanecer parado" },

            { value: 0, text: "Nenhuma vez" },
            { value: 1, text: "V√°rios dias" },
            { value: 2, text: "Mais da metade dos dias" },
            { value: 3, text: "Quase todos os dias" }
        ];

        let currentQuestion = 0;
        let answers = {};

        // Initialize test
        function initTest() {
            renderQuestion();
            updateProgress();
            updateNavigation();
        }

        // Render current question
        function renderQuestion() {
            const testForm = document.getElementById('testForm');
            const question = questions[currentQuestion];
            
            testForm.innerHTML = `
                <div class="question-card bg-white rounded-2xl p-8 ${answers[question.id] !== undefined ? 'answered' : ''}">
                    <div class="mb-6">
                        <span class="inline-block bg-orange-100 text-orange-800 text-sm font-medium px-3 py-1 rounded-full mb-4">
                            üò∞ Sintoma de Ansiedade
                        </span>
                        <h3 class="text-xl font-semibold text-gray-800 leading-relaxed">
                            ${currentQuestion + 1}. ${question.text}
                        </h3>
                        <p class="text-gray-600 mt-2 text-sm">
                            Nas √∫ltimas 2 semanas, com que frequ√™ncia voc√™ foi incomodado por este problema?
                        </p>
                    </div>
                    
                    <div class="space-y-3">
                        ${options.map(option => `
                            <label class="radio-option flex items-center p-4 rounded-lg cursor-pointer border border-gray-200 hover:bg-orange-50">
                                <input type="radio" name="question_${question.id}" value="${option.value}" 
                                       class="w-4 h-4 text-orange-600 border-gray-300 focus:ring-orange-500"
                                       ${answers[question.id] === option.value ? 'checked' : ''}
                                       onchange="handleAnswer(${question.id}, ${option.value})">
                                <span class="ml-3 text-gray-700 font-medium">${option.text}</span>
                            </label>
                        `).join('')}
                    </div>
                </div>
            `;
        }

        // Handle answer selection
        function handleAnswer(questionId, value) {
            answers[questionId] = value;
            updateProgress();
            updateNavigation();
            
            // Add visual feedback
            const questionCard = document.querySelector('.question-card');
            questionCard.classList.add('answered');
        }

        // Update progress bar
        function updateProgress() {
            const progress = (Object.keys(answers).length / questions.length) * 100;
            document.getElementById('progressBar').style.width = `${progress}%`;
            document.getElementById('progressText').textContent = `Pergunta ${currentQuestion + 1} de ${questions.length}`;
        }

        // Update navigation buttons
        function updateNavigation() {
            const prevBtn = document.getElementById('prevBtn');
            const nextBtn = document.getElementById('nextBtn');
            
            prevBtn.disabled = currentQuestion === 0;
            
            if (currentQuestion === questions.length - 1) {
                nextBtn.textContent = Object.keys(answers).length === questions.length ? 'üìä Ver Resultados' : 'Finalizar';
                nextBtn.disabled = answers[questions[currentQuestion].id] === undefined;
            } else {
                nextBtn.textContent = 'Pr√≥xima ‚Üí';
                nextBtn.disabled = answers[questions[currentQuestion].id] === undefined;
            }
        }

        // Navigation functions
        function prevQuestion() {
            if (currentQuestion > 0) {
                currentQuestion--;
                renderQuestion();
                updateProgress();
                updateNavigation();
            }
        }

        function nextQuestion() {
            if (currentQuestion < questions.length - 1) {
                currentQuestion++;
                renderQuestion();
                updateProgress();
                updateNavigation();
            } else {
                showResults();
            }
        }

        // Calculate and show results
        function showResults() {
            const totalScore = Object.values(answers).reduce((sum, value) => sum + value, 0);
            
            let severity = '';
            let color = '';
            let recommendation = '';
            let description = '';

            if (totalScore >= 0 && totalScore <= 4) {
                severity = 'Ansiedade M√≠nima';
                color = 'green';
                description = 'Sintomas de ansiedade dentro da normalidade';
                recommendation = 'Continue mantendo h√°bitos saud√°veis e t√©cnicas de relaxamento.';
            } else if (totalScore >= 5 && totalScore <= 9) {
                severity = 'Ansiedade Leve';
                color = 'yellow';
                description = 'Sintomas leves de ansiedade';
                recommendation = 'Considere t√©cnicas de relaxamento, exerc√≠cios e acompanhamento preventivo.';
            } else if (totalScore >= 10 && totalScore <= 14) {
                severity = 'Ansiedade Moderada';
                color = 'orange';
                description = 'Sintomas moderados de ansiedade';
                recommendation = 'Recomenda-se avalia√ß√£o com profissional de sa√∫de mental e poss√≠vel tratamento.';
            } else if (totalScore >= 15 && totalScore <= 21) {
                severity = 'Ansiedade Severa';
                color = 'red';
                description = 'Sintomas severos de ansiedade';
                recommendation = '√â importante buscar ajuda profissional imediatamente. Considere tratamento especializado.';
            }

            document.getElementById('testForm').classList.add('hidden');
            document.getElementById('navigationButtons').classList.add('hidden');
            
            const resultsSection = document.getElementById('resultsSection');
            const resultContent = document.getElementById('resultContent');
            
            resultContent.innerHTML = `

                    <div class="inline-block p-6 bg-${color}-100 rounded-full mb-4">
                        <span class="text-4xl">
                            ${color === 'green' ? 'üòå' : color === 'yellow' ? 'üòê' : color === 'orange' ? 'üòü' : 'üò∞'}
                        </span>
                    </div>
                    <h3 class="text-2xl font-bold text-gray-800 mb-2">${severity}</h3>
                    <p class="text-gray-600 mb-2">${description}</p>
                    <div class="text-3xl font-bold text-${color}-600">${totalScore}/21 pontos</div>
                </div>
                
                <div class="bg-${color}-50 p-6 rounded-xl mb-8">
                    <h4 class="text-lg font-semibold text-${color}-800 mb-3">üí° Recomenda√ß√µes</h4>
                    <p class="text-${color}-700">${recommendation}</p>
                </div>
                
                <div class="bg-gray-50 p-6 rounded-xl mb-6">
                    <h4 class="text-lg font-semibold text-gray-800 mb-4">üìä Interpreta√ß√£o dos Resultados</h4>

                        </div>
                        <div class="text-center p-3 bg-red-100 rounded-lg">

                    </div>
                    <div class="mt-4 p-3 bg-blue-50 rounded-lg">
                        <p class="text-sm text-blue-700">
                            <strong>Seu resultado:</strong> ${totalScore} pontos - ${severity}
                        </p>
                    </div>
                </div>
                
                <div class="bg-gray-50 p-6 rounded-xl">
                    <h4 class="text-lg font-semibold text-gray-800 mb-4">üìà An√°lise Detalhada</h4>

        // Reset test
        function resetTest() {
            currentQuestion = 0;
            answers = {};
            document.getElementById('testForm').classList.remove('hidden');
            document.getElementById('navigationButtons').classList.remove('hidden');
            document.getElementById('resultsSection').classList.add('hidden');
            initTest();
        }

        // Go to assessment center
        function goToCenter() {
            window.location.href = '/centro-avaliacao';
        }

        // Event listeners
        document.getElementById('prevBtn').addEventListener('click', prevQuestion);
        document.getElementById('nextBtn').addEventListener('click', nextQuestion);

        // Initialize the test
        initTest();
